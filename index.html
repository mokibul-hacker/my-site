async function checkKey() { const inputKey = document.getElementById("secret-input").value; const encoder = new TextEncoder(); const data = encoder.encode(inputKey); const hashBuffer = await crypto.subtle.digest('SHA-256', data); const hashArray = Array.from(new Uint8Array(hashBuffer)); const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join(''); // यह 'MOKIBUL2026' का सुरक्षित हैश (Hash) है const correctHash = "5e884898da28047151d0e56f8dc6292773603d0d6aabbdd62a11ef721d1542d8"; if (hashHex === correctHash) { alert("ACCESS GRANTED! स्वागत है मोकिबुल खान।"); window.location.href = "success.html"; } else { alert("WRONG KEY! कोशिश करते रहो।"); } }
